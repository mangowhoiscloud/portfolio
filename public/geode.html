# router.py — NL-first routing with offline fallback
async def nl_router(state: PipelineState) -> str:
    # Stage 1: LLM tool-use classification
    try:
        route = await llm.classify_route(state["user_input"])
        return route  # "full_pipeline" | "plan_mode" | "delegate" | ...
    except LLMError:
        pass

    # Stage 2: Offline regex fallback
    patterns = {
        r"계획|plan": "plan_mode",
        r"병렬|parallel|delegate": "delegate",
        r"평가|eval": "evaluation",
    }
    for pat, route in patterns.items():
        if re.search(pat, state["user_input"], re.I):
            return route

    # Stage 3: help fallback
    return "full_pipeline"  # safe default

    context = ConversationContext(max_turns=CONTEXT_WINDOW)
    round_count = 0

    while round_count < MAX_ROUNDS:
        response = await llm.messages(
            messages=context.get_messages(),
            tools=REGISTERED_TOOLS,  # 46 tools
        )
        if response.stop_reason == "end_turn":

        tool_results = await execute_tools(response.content)
        context.append(response, tool_results)
        round_count += 1

    return state.update(result=context.get_final_state())

async def dual_brain_verify(state: PipelineState) -> PipelineState:
    score_a, score_b = await asyncio.gather(
        claude_opus.evaluate(state),   # Primary
        gpt52.evaluate(state),         # Secondary
    )
    agreement = 1 - abs(score_a - score_b) / max(score_a, score_b)

    if agreement >= 0.67:
        return {"verification": "PASS", "final_score": (score_a + score_b) / 2}

    # Disagreement: trigger 3rd model
    score_c = await third_llm.evaluate(state)
    alpha = krippendorff_alpha([score_a, score_b, score_c])
    assert alpha >= 0.80, "Reliability below threshold"
    return {"verification": "PASS_3RD", "final_score": median([score_a, score_b, score_c])}

class SubAgentManager:
    async def delegate(self, ip_list: list[str]) -> list[AnalysisResult]:
        graph = TaskGraph(ip_list)  # build DAG
        runners = [
            IsolatedRunner(ip, self._build_context(ip))
            for ip in graph.resolve_order()

            *[self._run_with_sem(r, semaphore) for r in runners]
        )
        return CoalescingQueue(COALESCING_WINDOW_MS).flush(results)

Anthropic SDK를 래핑한 Claude 클라이언트입니다. Structured output과 exponential backoff 재시도 로직을 내장하여 안정적인 LLM 통신을 보장합니다. Claude client wrapping the Anthropic SDK. Ensures stable LLM communication with built-in structured output and exponential backoff retry logic.

ClaudeClient 구현 ClaudeClient Implementation 

class ClaudeClient:
    def __init__(self, config: GeodeConfig):
        self.client = Anthropic(api_key=config.anthropic_api_key)
        self.model = config.model_name
        self.max_retries = config.max_retries

    async def analyze(
        self,
        system_prompt: str,
        user_prompt: str,
        response_model: Type[BaseModel],
    ) -> BaseModel:
        for attempt in range(self.max_retries):
            try:
                response = self.client.messages.create(
                    model=self.model,
                    system=system_prompt,
                    messages=[{"role": "user", "content": user_prompt}],
                )
                return response_model.model_validate_json(
                    response.content[0].text
                )
            except Exception:
                await asyncio.sleep(2 ** attempt)
        raise MaxRetriesExceeded()
